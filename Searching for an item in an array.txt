//searching an item in an array

org 100
load addr
store next
input
store searchitem
clear
loop, load searchitem
addI next 				
skipcond 400
jump subtraction
halt

subtraction, clear
loadI next
subt searchitem
skipcond 400
jump increment
halt

increment, clear
load next  //increasing address of next
add one
store next

load count //decreasing the count
subt one
store count
skipcond 400
jump loop
load end
output


searchitem, dec 0
addr, hex 121
next, hex 0
count, dec 10 
one, dec 1
end, dec -1111
